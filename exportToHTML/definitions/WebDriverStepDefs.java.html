1    package definitions;
2    
3    import cucumber.api.java.en.And;
4    import cucumber.api.java.en.Given;
5    import cucumber.api.java.en.Then;
6    import cucumber.api.java.en.When;
7    
8    import org.openqa.selenium.By;
9    
10   import static org.assertj.core.api.AssertionsForClassTypes.assertThat;
11   import static support.TestContext.getDriver;
12   
13   public class WebDriverStepDefs {
14       @Given("I navigate{string}")
15       public void iNavigate(String sURL) {
16           switch (sURL){
17               case "google":
18                   getDriver().get("https://www.google.com/");
19                   case "duckduckgo":
20                   getDriver().get("https://www.duckduckgo.com/");
21   
22               case "swisscows":
23                   getDriver().get("https://www.swisscows.com/");
24               case "quote":
25                   getDriver().get("https://skryabin.com/market/quote.html");
26   
27               default:
28                   System.out.println("No URL found");
29   
30           }
31       }
32   
33       @And("I print page details")
34       public void iPrintPageDetails() {
35           System.out.println(getDriver().getTitle());
36           System.out.println(getDriver().getCurrentUrl());
37           System.out.println(getDriver().getWindowHandles());
38       }
39   
40       @When("I type {string} into email field")
41       public void iTypeIntoEmailField(String sEmail) {
42           getDriver().findElement(By.xpath("//input[@name='email']")).sendKeys(sEmail);
43       }
44   
45       @And("I click submit button")
46       public void iClickSubmitButton() {
47           getDriver().findElement(By.xpath("//button[@id='formSubmit']")).click();
48       }
49   
50   
51   
52       @Then("error message {string} should be displayed")
53       public void errorMessageShouldBeDisplayed(String sMessage) {
54           String sActualmessage =getDriver().findElement(By.xpath("//label[@id='email-error']")).getText();
55           System.out.println(sActualmessage);
56           assertThat(sActualmessage.equals(sMessage));
57       }
58   
59   
60       @When("I try validating date picker")
61       public void iTryValidatingDatePicker() {
62           getDriver().findElement(By.xpath("//input[@id='dateOfBirth']")).click();
63           getDriver().findElement(By.xpath("//select[@class='ui-datepicker-month']/option[5]")).click();
64           getDriver().findElement(By.xpath("//select[@class='ui-datepicker-year']/option[@value='2000']")).click();
65           getDriver().findElement(By.xpath("//td[@data-handler='selectDay']/a[text()='5']")).click();
66           String actualDate= getDriver().findElement(By.xpath("//input[@id='dateOfBirth']")).getAttribute("value");
67           assertThat(actualDate.equals("05/05/2000"));
68       }
69   }
